lemmatizer = WordNetLemmatizer()
stop_words = set(stopwords.words('english'))

def nltk_tokenizer_lemmatizer(text):
    text = text.lower()
    text = re.sub(r'[^a-z\s]', '', text)

    tokens = word_tokenize(text)

    processed_tokens = []
    for t in tokens:
        if len(t) > 1 and t not in stop_words:
            lemma = lemmatizer.lemmatize(t)
            processed_tokens.append(lemma)
    return processed_tokens

# TfidfVectorizer 설정 개선
tfidf_vectorizer = TfidfVectorizer(
    tokenizer=nltk_tokenizer_lemmatizer,
    lowercase=False,
    stop_words=None,
    token_pattern=None,
    ngram_range=(1, 2),

    min_df=1,
    max_df=0.8,
    max_features=150000,

    use_idf=True,
    smooth_idf=True,
    sublinear_tf=True
)

print("Training TfidfVectorizer on the corpus...")

def get_term_document_vector_with_fallback(term):
    """OOV 처리 포함 버전"""
    # 1차 시도: 정확한 매칭
    vec, key = get_term_document_vector(term)
    if vec is not None:
        return vec, key

    # 2차 시도: 부분 매칭 (모든 토큰의 평균)
    tokens = nltk_tokenizer_lemmatizer(term)
    if not tokens:
        return None, None

    vectors = []
    matched_keys = []

    for token in tokens:
        if token in vocab:
            term_idx = vocab[token]
            vec = tfidf_matrix[:, term_idx]
            vectors.append(vec)
            matched_keys.append(token)

    if vectors:
        # 평균 벡터 사용
        avg_vec = sum(vectors) / len(vectors)
        return avg_vec, f"avg({'+'.join(matched_keys)})"

    return None, None

# ----------------------------------------------

def get_term_document_vector(term):
    tokens = nltk_tokenizer_lemmatizer(term)
    if not tokens:
        return None, None

    # 여러 형태로 vocabulary 검색
    candidates = []

    if len(tokens) == 1:
        candidates = [tokens[0]]
    elif len(tokens) >= 2:
        candidates = [
            " ".join(tokens[:2]),
            " ".join(tokens),
            tokens[0]
        ]

    # Vocabulary에서 찾기
    for key in candidates:
        if key in vocab:
            term_idx = vocab[key]
            vec = tfidf_matrix[:, term_idx]
            return vec, key

    return None, None

            if vec1 is not None and vec2 is not None:
                sim_score = cosine_similarity(vec1.T, vec2.T)[0][0]

                # Partial match 카운트
                if 'avg(' in str(key1) or 'avg(' in str(key2):
                    partial_match_count += 1
                else:
                    matched_count += 1

                if index < 5:
                    print(f"  Similarity: {sim_score:.4f}")

                    zero_similarity_count += 1

                    sim_score = 0.0

                        missing.append(term2_raw)
